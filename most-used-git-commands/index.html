Most used git commands

Ben June 28, 2021 git

Setup

# The command saves the values in the global configuration file ~/.gitconfig
# set user name and user email
git config --global user.name "<your name>"
git config --global user.email "<yourname@email.com>"

# check configuration
git config --list

# set default editor to emacs (Linux)
git config --global core.editor emacs

# set default editor to emacs(Windows), assuming your emacs in installed in D: drive
git config --global core.editor "'D:/emacs-27.1-x86_64/bin/emacs.exe' -nw"

# set default editor to notepad++ (Windows only)
git config --global core.editor "'C:/Program Files/Notepad++/notepad++.exe' -multiInst -notabbar -nosession -noPlugin"
# Config Beyond Compare 4 as difftool

$ git config --global diff.tool bc4
$ git config --global difftool.prompt false
# Note: set parameters for Beyond Compare tool
$ git config --global difftool.bc4.cmd '"C:\Program Files\Beyond Compare 4\BCompare.exe" "$LOCAL" "$REMOTE"' 

# Config Becond Compare 4 as mergetool

$ git config --global merge.tool = bc4
$ git config --global mergetool.prompt false
$ git config --global mergetool.bc4.path 'C:\Program Files\Beyond Compare 4\BCompare.exe'

Branch


# Create a new branch called testbranch based on current branch and switch to the new branch
git checkout -b testbranch

# delete a local branch testbranch, current branch must not be testbranch
git branch -D testbranch

# delete remote branch testbranch
git push --delete origin testbranch

# rename local branch, rename current branch to newbranch
git branch -m newbranch

# rename local oldbranch to newbranch
git branch -m oldbranch newbranch

# rename remote branch, delete it first then push new brnach
git push origin --delete old_name
git push -u origin new_name

# create a new branch based on a tag
git checkout -b tags/v3.0 v3.0

# or like this:
git checkout -b newbranch v3.0
